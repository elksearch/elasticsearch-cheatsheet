### Linux / WSL2 — MANDATORY before starting

```bash
# Increase virtual memory (required for Elasticsearch)
sudo sysctl -w vm.max_map_count=262144

# Make it permanent across reboots
echo "vm.max_map_count=262144" | sudo tee -a /etc/sysctl.conf

# Disable swap
sudo swapoff -a

# Increase file descriptor limits
ulimit -n 65535
```

> **Docker Desktop (Mac/Windows):** Settings → Resources → Memory → set minimum **8 GB**

### docker-compose.yml

```yaml
services:

  # Runs once — generates TLS certs + sets passwords
  setup:
    image: docker.elastic.co/elasticsearch/elasticsearch:${STACK_VERSION}
    container_name: elk-setup
    volumes:
      - certs:/usr/share/elasticsearch/config/certs
    user: "0"
    command: >
      bash -c '
        if [ ! -f config/certs/ca.zip ]; then
          bin/elasticsearch-certutil ca --silent --pem -out config/certs/ca.zip;
          unzip config/certs/ca.zip -d config/certs;
        fi;
        if [ ! -f config/certs/certs.zip ]; then
          echo -ne "instances:\n  - name: elasticsearch\n    dns:\n      - elasticsearch\n      - localhost\n    ip:\n      - 127.0.0.1\n  - name: kibana\n    dns:\n      - kibana\n      - localhost\n    ip:\n      - 127.0.0.1\n" > config/certs/instances.yml;
          bin/elasticsearch-certutil cert --silent --pem -out config/certs/certs.zip \
            --in config/certs/instances.yml \
            --ca-cert config/certs/ca/ca.crt \
            --ca-key config/certs/ca/ca.key;
          unzip config/certs/certs.zip -d config/certs;
        fi;
        chown -R root:root config/certs;
        echo "Waiting for ES...";
        until curl -s --cacert config/certs/ca/ca.crt https://elasticsearch:9200 | grep -q "missing authentication"; do sleep 30; done;
        curl -s -X POST --cacert config/certs/ca/ca.crt \
          -u "elastic:${ELASTIC_PASSWORD}" -H "Content-Type: application/json" \
          https://elasticsearch:9200/_security/user/kibana_system/_password \
          -d "{\"password\":\"${KIBANA_PASSWORD}\"}";
        echo "Setup complete!";
      '
    healthcheck:
      test: ["CMD-SHELL", "[ -f config/certs/elasticsearch/elasticsearch.crt ]"]
      interval: 1s
      timeout: 5s
      retries: 120

  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:${STACK_VERSION}
    container_name: elasticsearch
    depends_on:
      setup:
        condition: service_healthy
    volumes:
      - certs:/usr/share/elasticsearch/config/certs
      - esdata:/usr/share/elasticsearch/data
    ports:
      - ${ES_PORT}:9200
    environment:
      - node.name=elasticsearch
      - cluster.name=${CLUSTER_NAME}
      - discovery.type=single-node
      - ELASTIC_PASSWORD=${ELASTIC_PASSWORD}
      - bootstrap.memory_lock=true
      - xpack.security.enabled=true
      - xpack.security.http.ssl.enabled=true
      - xpack.security.http.ssl.key=certs/elasticsearch/elasticsearch.key
      - xpack.security.http.ssl.certificate=certs/elasticsearch/elasticsearch.crt
      - xpack.security.http.ssl.certificate_authorities=certs/ca/ca.crt
      - xpack.security.transport.ssl.enabled=true
      - xpack.security.transport.ssl.key=certs/elasticsearch/elasticsearch.key
      - xpack.security.transport.ssl.certificate=certs/elasticsearch/elasticsearch.crt
      - xpack.security.transport.ssl.certificate_authorities=certs/ca/ca.crt
      - xpack.security.transport.ssl.verification_mode=certificate
      - xpack.license.self_generated.type=${LICENSE}
      - ES_JAVA_OPTS=${ES_JAVA_OPTS}
    mem_limit: ${ES_MEM_LIMIT}
    ulimits:
      memlock:
        soft: -1
        hard: -1
      nofile:
        soft: 65535
        hard: 65535
    healthcheck:
      test: ["CMD-SHELL", "curl -s --cacert config/certs/ca/ca.crt https://localhost:9200 | grep -q 'missing authentication'"]
      interval: 10s
      timeout: 10s
      retries: 120
    networks:
      - elk
    restart: unless-stopped

  kibana:
    image: docker.elastic.co/kibana/kibana:${STACK_VERSION}
    container_name: kibana
    depends_on:
      elasticsearch:
        condition: service_healthy
    volumes:
      - certs:/usr/share/kibana/config/certs
      - kibanadata:/usr/share/kibana/data
    ports:
      - ${KIBANA_PORT}:5601
    environment:
      - SERVERNAME=kibana
      - ELASTICSEARCH_HOSTS=https://elasticsearch:9200
      - ELASTICSEARCH_USERNAME=kibana_system
      - ELASTICSEARCH_PASSWORD=${KIBANA_PASSWORD}
      - ELASTICSEARCH_SSL_CERTIFICATEAUTHORITIES=config/certs/ca/ca.crt
      - XPACK_SECURITY_ENCRYPTIONKEY=${ENCRYPTION_KEY}
      - XPACK_ENCRYPTEDSAVEDOBJECTS_ENCRYPTIONKEY=${ENCRYPTION_KEY}
      - XPACK_REPORTING_ENCRYPTIONKEY=${ENCRYPTION_KEY}
    mem_limit: ${KB_MEM_LIMIT}
    healthcheck:
      test: ["CMD-SHELL", "curl -s -I http://localhost:5601 | grep -q 'HTTP/1.1 302'"]
      interval: 10s
      timeout: 10s
      retries: 120
    networks:
      - elk
    restart: unless-stopped

  logstash:
    image: docker.elastic.co/logstash/logstash:${STACK_VERSION}
    container_name: logstash
    depends_on:
      elasticsearch:
        condition: service_healthy
    volumes:
      - certs:/usr/share/logstash/config/certs
      - logstashdata:/usr/share/logstash/data
      - ./config/logstash/logstash.yml:/usr/share/logstash/config/logstash.yml:ro
      - ./config/logstash/pipeline:/usr/share/logstash/pipeline:ro
    ports:
      - ${LOGSTASH_BEATS_PORT}:5044
      - ${LOGSTASH_TCP_PORT}:5000
      - "9600:9600"
    environment:
      - ELASTIC_USER=elastic
      - ELASTIC_PASSWORD=${ELASTIC_PASSWORD}
      - ELASTIC_HOSTS=https://elasticsearch:9200
      - LS_JAVA_OPTS=${LS_JAVA_OPTS}
    mem_limit: ${LS_MEM_LIMIT}
    networks:
      - elk
    restart: unless-stopped

volumes:
  certs:
  esdata:
  kibanadata:
  logstashdata:

networks:
  elk:
    driver: bridge
```

## 🏗️ Part 2 — 3-Node Hot-Warm-Cold Cluster

### Architecture

```
┌─────────────────────────────────────────────────────────┐
│                Elasticsearch Cluster                     │
│                                                          │
│  ┌────────────┐  ┌────────────┐  ┌────────────────────┐ │
│  │    es01    │  │    es02    │  │       es03         │ │
│  │  HOT tier  │  │  WARM tier │  │  COLD/FROZEN tier  │ │
│  │  master +  │  │  master +  │  │  master +          │ │
│  │  data_hot  │  │  data_warm │  │  data_cold/frozen  │ │
│  │  :9200     │  │            │  │                    │ │
│  └────────────┘  └────────────┘  └────────────────────┘ │
└─────────────────────────────────────────────────────────┘
         ▲
    ┌────┴────┐
    │  Kibana │
    │  :5601  │
    └─────────┘
```

### Node Roles

| Node | Roles | Tier | Storage |
|---|---|---|---|
| `es01` | master, data_hot, data_content, ingest | HOT | SSD — active indexing |
| `es02` | master, data_warm, data_content | WARM | HDD — 30-90 day data |
| `es03` | master, data_cold, data_frozen | COLD | Cheapest — 90+ days |

## 📦 Part 3 — Filebeat + Metricbeat + Elastic Agent + Fleet + APM

### docker-compose-beats.yml

```yaml
# Use WITH docker-compose.yml:
# docker compose -f docker-compose.yml -f docker-compose-beats.yml up -d

services:

  # ── FILEBEAT — lightweight log shipper ──────────────────────
  filebeat:
    image: docker.elastic.co/beats/filebeat:${STACK_VERSION}
    container_name: filebeat
    user: root
    depends_on:
      elasticsearch:
        condition: service_healthy
    volumes:
      - certs:/usr/share/filebeat/certs
      - filebeatdata:/usr/share/filebeat/data
      - ./config/filebeat/filebeat.yml:/usr/share/filebeat/filebeat.yml:ro
      - /var/lib/docker/containers:/var/lib/docker/containers:ro
      - /var/log:/var/log:ro
      - /var/run/docker.sock:/var/run/docker.sock:ro
    environment:
      - ELASTIC_USER=elastic
      - ELASTIC_PASSWORD=${ELASTIC_PASSWORD}
      - ELASTIC_HOSTS=https://elasticsearch:9200
      - KIBANA_HOSTS=http://kibana:5601
    command: ["--strict.perms=false"]
    networks:
      - elk
    restart: unless-stopped

  # ── METRICBEAT — system & service metrics ───────────────────
  metricbeat:
    image: docker.elastic.co/beats/metricbeat:${STACK_VERSION}
    container_name: metricbeat
    user: root
    depends_on:
      elasticsearch:
        condition: service_healthy
    volumes:
      - certs:/usr/share/metricbeat/certs
      - metricbeatdata:/usr/share/metricbeat/data
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - /sys/fs/cgroup:/hostfs/sys/fs/cgroup:ro
      - /proc:/hostfs/proc:ro
      - /:/hostfs:ro
    environment:
      - ELASTIC_USER=elastic
      - ELASTIC_PASSWORD=${ELASTIC_PASSWORD}
      - ELASTIC_HOSTS=https://elasticsearch:9200
      - KIBANA_HOSTS=http://kibana:5601
    command: ["--strict.perms=false", "-system.hostfs=/hostfs"]
    networks:
      - elk
    restart: unless-stopped

  # ── FLEET SERVER — manages all Elastic Agents ───────────────
  fleet-server:
    image: docker.elastic.co/beats/elastic-agent:${STACK_VERSION}
    container_name: fleet-server
    user: root
    depends_on:
      elasticsearch:
        condition: service_healthy
      kibana:
        condition: service_healthy
    volumes:
      - certs:/certs
      - fleetserverdata:/usr/share/elastic-agent
    ports:
      - ${FLEET_PORT:-8220}:8220
    environment:
      - FLEET_SERVER_ENABLE=true
      - FLEET_SERVER_HOST=0.0.0.0
      - FLEET_SERVER_PORT=8220
      - FLEET_SERVER_ELASTICSEARCH_HOST=https://elasticsearch:9200
      - FLEET_SERVER_ELASTICSEARCH_CA=/certs/ca/ca.crt
      - FLEET_SERVER_SERVICE_TOKEN=${FLEET_SERVER_SERVICE_TOKEN}
      - FLEET_SERVER_POLICY_ID=fleet-server-policy
      - ELASTICSEARCH_HOST=https://elasticsearch:9200
      - ELASTICSEARCH_USERNAME=elastic
      - ELASTICSEARCH_PASSWORD=${ELASTIC_PASSWORD}
      - ELASTICSEARCH_CA=/certs/ca/ca.crt
      - KIBANA_HOST=http://kibana:5601
      - KIBANA_USERNAME=elastic
      - KIBANA_PASSWORD=${ELASTIC_PASSWORD}
    healthcheck:
      test: ["CMD-SHELL", "curl -s -k https://localhost:8220/api/status | grep -q 'HEALTHY'"]
      interval: 15s
      timeout: 10s
      retries: 30
    networks:
      - elk
    restart: unless-stopped

  # ── ELASTIC AGENT — universal collector managed by Fleet ────
  elastic-agent:
    image: docker.elastic.co/beats/elastic-agent:${STACK_VERSION}
    container_name: elastic-agent
    user: root
    depends_on:
      fleet-server:
        condition: service_healthy
    volumes:
      - certs:/certs
      - elasticagentdata:/usr/share/elastic-agent
      - /var/lib/docker/containers:/var/lib/docker/containers:ro
      - /var/log:/var/log:ro
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - /sys/fs/cgroup:/hostfs/sys/fs/cgroup:ro
      - /proc:/hostfs/proc:ro
    environment:
      - FLEET_URL=https://fleet-server:8220
      - FLEET_CA=/certs/ca/ca.crt
      - FLEET_ENROLLMENT_TOKEN=${FLEET_ENROLLMENT_TOKEN}
      - ELASTICSEARCH_HOST=https://elasticsearch:9200
      - ELASTICSEARCH_USERNAME=elastic
      - ELASTICSEARCH_PASSWORD=${ELASTIC_PASSWORD}
      - ELASTICSEARCH_CA=/certs/ca/ca.crt
    networks:
      - elk
    restart: unless-stopped

  # ── APM SERVER — application performance monitoring ─────────
  apm-server:
    image: docker.elastic.co/apm/apm-server:${STACK_VERSION}
    container_name: apm-server
    depends_on:
      elasticsearch:
        condition: service_healthy
      kibana:
        condition: service_healthy
    volumes:
      - certs:/usr/share/apm-server/certs
    ports:
      - ${APM_PORT:-8200}:8200
    command: >
      apm-server -e
        -E apm-server.host=0.0.0.0:8200
        -E apm-server.secret_token=${APM_SECRET_TOKEN}
        -E output.elasticsearch.hosts=["https://elasticsearch:9200"]
        -E output.elasticsearch.username=elastic
        -E output.elasticsearch.password=${ELASTIC_PASSWORD}
        -E output.elasticsearch.ssl.certificate_authorities=["/usr/share/apm-server/certs/ca/ca.crt"]
        -E setup.kibana.host=http://kibana:5601
        -E apm-server.rum.enabled=true
    healthcheck:
      test: ["CMD-SHELL", "curl -s http://localhost:8200/ | grep -q 'build_date'"]
      interval: 10s
      timeout: 5s
      retries: 30
    networks:
      - elk
    restart: unless-stopped

volumes:
  certs:
```

## ❌ Part 7 — Common Errors & Fixes

| Error | Cause | Fix |
|---|---|---|
| `vm.max_map_count too low` | Linux kernel setting | `sudo sysctl -w vm.max_map_count=262144` |
| `memory locking failed` | Missing ulimits | Add `ulimits.memlock: soft: -1, hard: -1` in compose |
| `Kibana server not ready` | Startup takes time | Wait 3 mins, check `docker compose logs kibana` |
| `Authentication failed` | Wrong password | Reset: `docker compose exec elasticsearch bin/elasticsearch-reset-password -u elastic` |
| Cluster RED | Unassigned shards | `POST _cluster/reroute?retry_failed=true` |
| `Disk watermark exceeded` | Low disk space | Free disk or increase watermark threshold |
| Logstash can't connect | SSL/auth issue | Check `docker compose logs logstash`, verify certs |
| Filebeat permission denied | Not running as root | Set `user: root` in compose |
| APM can't connect | Wrong token | Verify `APM_SECRET_TOKEN` matches agent config |
| Fleet agent unhealthy | Token expired | Generate new enrollment token in Kibana |

## ✅ Part 8 — Production Checklist

```
Security:
  □ Change ALL passwords in .env
  □ Enable TLS on all services (included in this setup)
  □ Never use elastic user for applications — create RBAC users
  □ Don't expose :9200 directly — put nginx/HAProxy in front
  □ Enable audit logging
  □ Use secrets manager instead of .env in production

Performance:
  □ ES_JAVA_OPTS = 50% of RAM, max 30GB per node
  □ bootstrap.memory_lock=true + configure ulimits
  □ vm.max_map_count=262144 permanent in /etc/sysctl.conf
  □ SSD storage for hot nodes
  □ swapoff -a on all ES nodes

Reliability:
  □ Minimum 3 dedicated master nodes
  □ At least 1 replica per index
  □ ILM policy configured for all indices
  □ Automated snapshots to S3/GCS
  □ Kibana alerting for cluster health RED/YELLOW

Operations:
  □ Pin exact STACK_VERSION — never use "latest"
  □ Test upgrades in staging first
  □ Document backup & restore procedures
  □ Configure Docker log rotation
  □ Enable AutoOps (free since Feb 2026)
```
